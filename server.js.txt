const express = require("express");
const axios = require("axios");

const app = express();
app.use(express.json());

app.post("/webhook", async (req, res) => {
  try {
    const data = req.body;

    const orderId = data.name;
    const phone = data.shipping_address.phone;
    const customerName = data.shipping_address.first_name;

    const tracking = data.fulfillments?.[0]?.tracking_number;

    if (!tracking) return res.send("No AWB yet");

    await axios.post("https://api.interakt.ai/v1/public/message/", {
      countryCode: "+91",
      phoneNumber: phone.replace("+91", ""),
      type: "Template",
      template: {
        name: "order_dispatch",
        languageCode: "en",
        bodyValues: [
          customerName,
          orderId,
          tracking
        ]
      }
    }, {
      headers: {
        Authorization: "Bearer YOUR_API_KEY"
      }
    });

    res.send("Message Sent ✅");
  } catch (err) {
    console.log(err);
    res.status(500).send("Error");
  }
});

app.listen(3000, () => console.log("Server running 🚀"));
